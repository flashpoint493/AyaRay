216  void setEuler(const float& yaw, const float& pitch, const float& roll)
217  {

256  void getEulerZYX(float &yaw_z, float &pitch_y, float &roll_x) const
257  {
258  float squ = m_val[0] * m_val[0];
259  float sqx = m_val[1] * m_val[1];
260  float sqy = m_val[2] * m_val[2];
261  float sqz = m_val[3] * m_val[3];
262  float sarg = -2.f * (m_val[0] * m_val[2] - m_val[3] * m_val[1]);
263 
264  // If the pitch angle is PI/2 or -PI/2, we can only compute
265  // the sum roll + yaw. However, any combination that gives
266  // the right sum will produce the correct orientation, so we
267  // set rollX = 0 and compute yawZ.
268  if (sarg <= -0.99999f)
269  {
270  pitch_y = -0.5f * float(M_PI);
271  roll_x = 0;
272  yaw_z = 2.f * atan2f(m_val[0], -m_val[1]);
273  }
274  else if (sarg >= 0.99999f)
275  {
276  pitch_y = 0.5f * float(M_PI);
277  roll_x = 0;
278  yaw_z = 2.f * atan2f(-m_val[0], m_val[1]);
279  }
280  else
281  {
282  pitch_y = asinf(sarg);
283  roll_x = atan2f(2.f * (m_val[1] * m_val[2] + m_val[3] * m_val[0]), squ - sqx - sqy + sqz);
284  yaw_z = atan2f(2.f * (m_val[0] * m_val[1] + m_val[3] * m_val[2]), squ + sqx - sqy - sqz);
285  }

635  Quaternion slerp(const Quaternion &q, const float &t) const
636  {
637  const float magnitude = Sqrt(length2() * q.length2());
638  assert(magnitude > 0.f);
639 
640  const float product = dot(q) / magnitude;
641  const float absproduct = fabsf(product);
642 
643  if (absproduct < 1.0f - SIMD_EPSILON) {
644  // Take care of long angle case see http://en.wikipedia.org/wiki/Slerp
645  const float theta = cosf(absproduct);
646  const float d = sinf(theta);
647  assert(d > 0.f);
648 
649  const float sign = (product < 0) ? -1.f : 1.f;
650  const float s0 = sinf((1.0f - t) * theta) / d;
651  const float s1 = sinf(sign * t * theta) / d;
652 
653  return Quaternion(
654  (m_val[0] * s0 + q.x() * s1),
655  (m_val[1] * s0 + q.y() * s1),
656  (m_val[2] * s0 + q.z() * s1),
657  (m_val[3] * s0 + q.w() * s1));
658  }
