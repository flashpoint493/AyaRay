100  for (uint32_t i = 0; i < MT19937_BUFF_LENGTH; i++) {
101  int y = (MT[i] & 0x80000000) // bit 31 (32nd bit) of MT[i]
102  + (MT[(i + 1) % 624] & 0x7fffffff); // bits 0-30 (first 31 bits) of MT[...]
103  MT[i] = MT[(i + 397) % 624] ^ (y >> 1);
104  if (!(y & 1)) {
105  MT[i] = MT[i] ^ 2567483615u; // 2567483615 == 0x9908b0df
106  }
107  }
108  }

115  MT[0] = (uint32_t)seed;
116  idx = 0;
117  for (uint32_t i = 1; i < MT19937_BUFF_LENGTH; i++) {
118  MT[i] = 1812433253u * (MT[i - 1] ^ (MT[i - 1] >> 30)) + i; // 1812433253 == 0x6c078965
119  }

121  inline uint32_t rand32() {
